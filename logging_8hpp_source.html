252inline std::shared_ptr<spdlog::logger> get(const std::string &name) {
253 std::shared_ptr<spdlog::logger> logger = spdlog::get(name);
254 if (logger == nullptr) {
255 if (sink != nullptr)
256 logger = std::make_shared<spdlog::logger>(name, sink);
257 else
258 logger = std::make_shared<spdlog::logger>(name);
259 logger->set_level(spdlog::level::level_enum::trace);
260 spdlog::register_logger(logger);
261 }
262 return logger;
263}
